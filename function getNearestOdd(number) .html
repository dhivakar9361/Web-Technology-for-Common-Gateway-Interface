function getNearestOdd(number) { // Check if the number is already odd if (number % 2 !== 0) { return number; } else { // If even, get the nearest odd return number + 1; } } // Prompt the user to enter a number const userInput = prompt("Enter a number:"); const number = parseInt(userInput); if (!isNaN(number)) { const nearestOdd = getNearestOdd(number); alert("The nearest odd number is: " + nearestOdd); } else { alert("Please enter a valid number."); }
