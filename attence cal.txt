    <script>
        function calculateAttendance() {
            const totalClasses = parseInt(document.getElementById("totalClasses").value);
            const attendedClasses = parseInt(document.getElementById("attendedClasses").value);
            const resultDiv = document.getElementById("result");

            if (isNaN(totalClasses) || isNaN(attendedClasses) || totalClasses <= 0) {
                resultDiv.innerHTML = "Please enter valid numbers for both fields.";
                resultDiv.classList.remove("warning");
                return;
            }

            const attendancePercentage = (attendedClasses / totalClasses) * 100;
            resultDiv.innerHTML = `Your attendance is ${attendancePercentage.toFixed(2)}%.`;

            if (attendancePercentage < 75) {
                resultDiv.innerHTML += "<br><span class='warning'>Warning: Your attendance is below the required 75%.</span>";
            } else {
                resultDiv.classList.remove("warning");
            }
        }
    </script>
